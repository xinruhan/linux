# 解压FRP Unzip FRP

mv frp_0.15.1_linux_amd64 frp
# 设置frps.ini setting frps.ini
cd /frp/ #path of frps.ini
nano frps.ini

[common]
bind_port = 7000

dashboard_port = 7500 # optional
